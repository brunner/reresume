% -------------------- Commands --------------------

% Produces the header for the resume.
% @param 1: header line
% @param 2: subheader line
\newcommand{\header}[2]{
  \begin{center}
    {\LARGE \textbf{#1}} \\
    \vspace*{5px}
    {#2} \\
  \end{center}
}

% Produces a horizontal rule.
\newcommand{\hr}{
  \noindent \makebox[\linewidth]{
    \rule{\textwidth}{1pt}
  } 
}

% Produces underlined text.
% h/t https://alexwlchan.net/2017/10/latex-underlines/
% @param 1: text
\newcommand{\myuline}[1]{%
  \color{black!80}{\uline{\phantom{#1}}}%
  \llap{\contour{white}{#1}}%
}

% Produces a section header.
% @param 1: section header text
\renewcommand{\section}[1]{
  \vspace*{5px}
  \noindent {\Large \textbf{#1}} \\
  \vspace*{-24px} \\
  \hr
  \vspace*{-10px}
}

% Produces a section subheader.
% @param 1: section subheader text
\renewcommand{\subsection}[1]{
  \noindent {\large \textbf{#1}} \\
}

% Produces a section subheader detail.
% @param 1: section subheader detail text
\newcommand{\subsectiondetail}[2]{
  \noindent {\textit{#1} \hfill {#2}} \\
}

% Produces indented text.
% @param 1: text
\newcommand{\listitem}[1]{
  $\bullet$ {#1} \\
}

% Produces the footer for the resume.
% @param 1: footer line 1
\newcommand{\footer}[1]{
  \begin{center}
    {#1}
  \end{center}
}

% Produces a twiddle character.
\newcommand{\twiddle}{\raise.2ex\hbox{$\scriptstyle\sim$}}
